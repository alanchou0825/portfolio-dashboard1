function doGet(e) {
  var sheetName = (e && e.parameter && e.parameter.sheet) ? e.parameter.sheet : 'holdings';
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet || sheet.getLastRow() < 2) return respond({holdings:[]});
  var rows = sheet.getRange(2,1,sheet.getLastRow()-1,5).getValues();
  var holdings = rows.filter(function(r){return r[0];}).map(function(r){
    return {market:String(r[0]),name:String(r[1]),code:String(r[2]),shares:Number(r[3]),cost:Number(r[4])};
  });
  return respond({holdings:holdings});
}
function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var sheetName = data.sheet || 'holdings';
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
  sheet.clearContents();
  sheet.appendRow(['市場','名稱','代號','股數','成本']);
  data.holdings.forEach(function(h){sheet.appendRow([h.market,h.name,h.code,h.shares,h.cost]);});
  return respond({ok:true,saved:data.holdings.length});
}
function respond(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

黃金交叉
// MA5 穿越 MA20 向上 · 做多訊號
定義
短期均線（MA5）由下往上穿越長期均線（MA20），代表短期動能強於長期趨勢，市場轉為多方主導。
操作參考
交叉確認當日或次日開盤進場做多，搭配成交量放大效果更佳。止損設在 MA20 下方。
關鍵觀察
訊號出現後若 MA5 持續走揚且兩線間距擴大，趨勢確立；若交叉後快速回頭，需警惕假突破。
死亡交叉
// MA5 穿越 MA20 向下 · 出場警示
定義
短期均線（MA5）由上往下穿越長期均線（MA20），代表短期賣壓加重，多方力量減弱，市場可能轉空。
操作參考
死亡交叉出現時考慮減碼或出場，避免持續跟跌。可搭配 RSI 確認是否已超賣再決定是否留倉。
關鍵觀察
在下跌趨勢中死亡交叉後出現量縮反彈，需留意反彈是否持續；若無量，通常是逃命波而非底部。
RSI 指標
// 相對強弱指數 · 30 超賣 / 70 超買 · 背離訊號
定義
RSI 衡量一段期間內漲幅與跌幅的相對強度，數值介於 0–100。RSI＞70 為超買，RSI＜30 為超賣。
操作參考
RSI 跌破 30 後回升穿越 30 可視為買點；RSI 突破 70 後回落跌穿 70 可視為賣點。
關鍵觀察
當股價創新高但 RSI 未創高（頂背離），是趨勢反轉的強力警示；股價創低但 RSI 未創低（底背離），可能即將反彈。
支撐壓力
// 水平支撐線與壓力線 · 震盪區間示意
定義
支撐線是價格多次下跌後反彈的低點連線；壓力線是多次上漲後回落的高點連線，兩者構成震盪區間。
操作參考
價格接近支撐線時考慮買進；接近壓力線時考慮獲利了結。若放量突破壓力，壓力轉支撐，可加碼。
關鍵觀察
支撐或壓力被有效突破的關鍵在於「放量」。無量突破通常是假突破，需等收盤確認再行動。
量價關係
// 股價 + 成交量 · 爆量突破螢光色標示
定義
量價同步上漲是強勢訊號；價漲量縮代表動能衰竭；價跌量縮代表賣壓不重；價跌量增代表主力出貨。
操作參考
爆量突破前高（螢光色柱）是強力買進訊號。量縮回測不破前高支撐，是加碼的好時機。
關鍵觀察
長期量縮整理後突然爆量，往往是主力進場的訊號；但需搭配價格是否突破關鍵壓力才能確認。
